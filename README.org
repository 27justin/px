#+TITLE: phoenix lang

Statically compiled programming language.

# Example

#+BEGIN_SRC px

rgb := type (r: i8, g: i8, b: i8)

rectangle := struct {
  size: (f32, f32)
}

circle := struct {
  radius: f32
}

drawable := contract {
  draw: fn (!self, color: rgb)
}

draw := fn (obj: drawable, color: rgb) {
obj.draw(color)
}
